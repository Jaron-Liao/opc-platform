opc_agent.py 
from opc import Agent, Tool

# 初始化OPC Agent
agent = Agent(
    model="qwen-max",
    system="你是OPC影视创作助手"
)

# 注册工具
@agent.tool
def generate_script(theme: str) -> str:
    """根据主题生成影视脚本"""
    return agent.complete(
        f"为'{theme}'创作短片脚本"
    )

# 运行任务
result = agent.run(
    "生成一个科幻短片概念"
)
print(result.output)
